import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

df = pd.DataFrame({"income": [5, 8, 10, 12, 15, 20, 300, 500, 800]})

sns.histplot(df["income"], kde=True)
plt.show()

print("Skewness:", df["income"].skew())
